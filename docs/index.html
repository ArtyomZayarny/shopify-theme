price-vat.liquid 
<!-- Metafield guard + bidirectional VAT -->
{%- if settings.vat_enabled
    and product.metafields.custom.show_tax.value == true -%}

  {%- assign vat_rate = settings.vat_rate -%}

  {%- if settings.vat_prices_include_tax -%}
    {%- comment -%} excl = price × 100 / (100 + rate) {%- endcomment -%}
    {%- assign divisor = 100 | plus: vat_rate -%}
    {%- assign excl_price = price | times: 100
                                | divided_by: divisor -%}
  {%- else -%}
    {%- comment -%} incl = price × (100 + rate) / 100 {%- endcomment -%}
    {%- assign multiplier = 100 | plus: vat_rate -%}
    {%- assign incl_price = price | times: multiplier
                                 | divided_by: 100 -%}
  {%- endif -%}

  <div class="price-vat">
    Incl. VAT: {{ incl_price | money }}
    Excl. VAT: {{ excl_price | money }}
  </div>
{%- endif -%}
product-tabs.js 
if (!customElements.get('product-tabs')) {
  customElements.define('product-tabs',
    class ProductTabs extends HTMLElement {
      connectedCallback() {
        this.buttons = Array.from(
          this.querySelectorAll('[role="tab"]'));
        this.panels = Array.from(
          this.querySelectorAll('[role="tabpanel"]'));

        this.buttons.forEach((btn, i) => {
          btn.setAttribute('aria-selected',
            i === 0 ? 'true' : 'false');
          btn.addEventListener('click',
            () => this.activateTab(i));
          btn.addEventListener('keydown',
            (e) => this.handleKeydown(e, i));
        });
      }

      handleKeydown(event, currentIndex) {
        const { key } = event;
        let newIndex;
        switch (key) {
          case 'ArrowRight':
            newIndex = (currentIndex + 1)
              % this.buttons.length; break;
          case 'ArrowLeft':
            newIndex = (currentIndex - 1
              + this.buttons.length)
              % this.buttons.length; break;
          case 'Home': newIndex = 0; break;
          case 'End':
            newIndex = this.buttons.length - 1;
            break;
          default: return;
        }
        event.preventDefault();
        this.activateTab(newIndex);
      }
    }
  );
}
